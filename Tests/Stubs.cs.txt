namespace AIFrontier.Models
{
public sealed class NewsEdition
{
    public string EditionDate { get; set; } = string.Empty;
    public List<NewsItem> Items { get; set; } = [];
}

public sealed class NewsItem
{
    public string Id { get; set; } = string.Empty;
    public string Category { get; set; } = string.Empty;
    public string ContentType { get; set; } = string.Empty;
    public List<string> Topics { get; set; } = [];
    public string Title { get; set; } = string.Empty;
    public string Summary { get; set; } = string.Empty;
    public string ReaderContext { get; set; } = string.Empty;
    public List<TermExplanation> TermExplanations { get; set; } = [];
    public List<string> Tags { get; set; } = [];
    public string PublishedAt { get; set; } = string.Empty;
    public string SourceName { get; set; } = string.Empty;
    public string SourceUrl { get; set; } = string.Empty;
    public double HotScore { get; set; }
    public double SourceQualityScore { get; set; }
    public double InnovationScore { get; set; }
    public double TechnicalRelevanceScore { get; set; }
    public double FreshnessScore { get; set; }
    public bool IsPrimarySourceVerified { get; set; }
    public DateTimeOffset? AddedAt { get; set; }
    public bool IsRead { get; set; }
    public bool HasRecentArrival => AddedAt is { } added && DateTimeOffset.Now - added <= TimeSpan.FromDays(3);
    public bool IsNew => HasRecentArrival && !IsRead;
}


public sealed class TermExplanation
{
    public string Term { get; set; } = string.Empty;
    public string Explanation { get; set; } = string.Empty;
}

public sealed class PreferenceProfile
{
    public int SchemaVersion { get; set; } = 1;
    public DateTimeOffset UpdatedAt { get; set; } = DateTimeOffset.Now;
    public Dictionary<string, double> TopicWeights { get; set; } = [];
    public Dictionary<string, double> SourceWeights { get; set; } = [];
    public string Depth { get; set; } = "入门解释";
    public string PromptHint { get; set; } = string.Empty;
    public List<string> BlockedSources { get; set; } = [];
    public List<string> BlockedTopics { get; set; } = [];
    public Dictionary<string, ArticlePreference> ArticlePreferences { get; set; } = [];
    public List<string> BookmarkedIds { get; set; } = [];
    public int ExplicitFeedbackCount { get; set; }
}

public sealed class ArticlePreference
{
    public bool DetailOpened { get; set; }
    public bool SourceOpened { get; set; }
    public bool CodexOpened { get; set; }
    public bool IsBookmarked { get; set; }
    public string Reaction { get; set; } = string.Empty;
    public double? Rating { get; set; }
}

public sealed record FeedbackEvent(
    string NewsId,
    string Action,
    double? Score,
    string Category,
    string Source,
    DateTimeOffset CreatedAt);

public sealed record PreferenceRecordResult(PreferenceProfile Profile, bool Persisted);

public enum NewsUpdateMode { CloudPreferred, LocalCodexOnly }
public sealed class LocalNewsUpdateSettings
{
    public NewsUpdateMode Mode { get; set; }
    public bool LocalFallbackEnabled { get; set; } = true;
    public string LastStatus { get; set; } = string.Empty;
}
public sealed record DailyNewsUpdateResult(bool Checked, bool NewsChanged, bool UsedLocalCodex, bool CodexConnected, string Status);
public sealed record ReadyNewsPromotionResult(int PublishedCount, int RemainingCount, bool NewsChanged, string Status);
}

namespace AIFrontier.Services
{
using AIFrontier.Models;
public sealed class NewsService
{
    public Task<NewsEdition> LoadAsync(bool refreshRemote = true) => Task.FromResult(new NewsEdition());
}

public sealed class DailyNewsUpdateCoordinator
{
    public Task<LocalNewsUpdateSettings> LoadSettingsAsync(CancellationToken cancellationToken = default) =>
        Task.FromResult(new LocalNewsUpdateSettings());
    public Task<int> GetReadyCountAsync(CancellationToken cancellationToken = default) => Task.FromResult(0);
    public Task SavePreferencesAsync(bool personalMode, bool fallbackEnabled, CancellationToken cancellationToken = default) => Task.CompletedTask;
    public Task<DailyNewsUpdateResult> RunAsync(NewsEdition edition, bool force, Action<string>? reportStatus = null, CancellationToken cancellationToken = default) =>
        Task.FromResult(new DailyNewsUpdateResult(true, false, false, false, string.Empty));
    public Task<ReadyNewsPromotionResult> PublishReadyAsync(NewsEdition edition, int count, CancellationToken cancellationToken = default) =>
        Task.FromResult(new ReadyNewsPromotionResult(0, 0, false, string.Empty));
}

public static class ReadyNewsQueueService
{
    public const int PublishBatchSize = 3;
}

public static class NewsOrderingPolicy
{
    public static List<NewsItem> PrioritizeArrivalState(IEnumerable<NewsItem> items) => items.ToList();
}

public enum UpdateChoice { InstallNow, EnableAutoUpdate, SkipVersion, NeverPrompt }

public sealed class UpdateCheckResult
{
    public string Status { get; set; } = string.Empty;
}

public sealed class UpdateService
{
    public Task<UpdateCheckResult> CheckAsync() => Task.FromResult(new UpdateCheckResult());
    public Task<string> ApplyChoiceAsync(UpdateCheckResult update, UpdateChoice choice) => Task.FromResult(string.Empty);
    public Task<string> ResetPromptPreferencesAsync() => Task.FromResult(string.Empty);
}

public sealed class CodexResult
{
    public string Status { get; set; } = string.Empty;

}

public sealed class CodexIntegrationService
{
    public string WorkspacePath => string.Empty;

    public Task<CodexResult> ConnectAsync() => Task.FromResult(new CodexResult());
    public Task<CodexResult> AnalyzeAsync(NewsItem item) => Task.FromResult(new CodexResult());
}
